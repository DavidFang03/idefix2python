Particle quantities over heatmap¶

 1from idefix2python import RunContext, Pipeline, PartQuantity, MapMovie2D
 2from pathlib import Path
 3import numpy as np
 4
 5projectPath = Path(__file__).parent / "data_test"
 6task = "multipart_2D"
 7configPath = projectPath / "config.json"
 8# By default the vtks are expected to be in {projetPath}/{task}/outputs/vtks/
 9
10
11custom_fields2D = [
12    MapMovie2D(
13        "RHO",
14        r"$\rho$",
15        plot_coords=[0, 0],
16        title="Density",
17        streamlines=["VX1", "VX2"],
18        uids="all",
19    ),
20    MapMovie2D(
21        "VX1",
22        r"$v_x$",
23        plot_coords=[0, 1],
24        streamlines=["VX1", "VX2"],
25        uids=[10, 15],
26    ),
27]
28
29
30def z(partvtk):
31    r = partvtk.data["PART_X1"]
32    theta = partvtk.data["PART_X2"]
33    return r * np.cos(theta)
34
35
36pqs = [
37    PartQuantity(
38        "PART_X1",
39        "PART_X1",
40        uids="all",
41        plot_coords=[0, 0],
42    ),
43    PartQuantity(
44        "PART_X2",
45        "PART_X2",
46        uids="all",
47        plot_coords=[0, 1],
48    ),
49    PartQuantity("z", "z", uids="all", plot_coords=[0, 2], compute=z),
50]
51
52
53# Initialize context
54runContext = RunContext(task, projectPath, configPath=configPath)
55
56pipeline = Pipeline(runContext, partQuantities=pqs, movies2D=custom_fields2D)
57
58pipeline.run()
